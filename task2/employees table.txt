CREATE TABLE employees (
    id INTEGER PRIMARY KEY,
    name TEXT NOT NULL,
    department TEXT,
    salary INTEGER DEFAULT 30000,
    joining_date TEXT
);

-- INSERT statements
INSERT INTO employees (id, name, department, salary, joining_date)
VALUES (1, 'Alice', 'HR', 40000, '2023-01-15');

INSERT INTO employees (id, name, department, salary, joining_date)
VALUES (2, 'Bob', NULL, 35000, '2023-02-01');  -- department is NULL

INSERT INTO employees (id, name)
VALUES (3, 'Charlie');  -- department, salary, date will take default or NULL

-- UPDATE statements
UPDATE employees
SET department = 'IT'
WHERE name = 'Bob';

UPDATE employees
SET salary = 50000
WHERE id = 3;

-- DELETE statement
DELETE FROM employees
WHERE id = 1;

-- Handling multiple rows
UPDATE employees
SET salary = salary + 1000
WHERE department = 'IT';

-- Rollback demonstration (Only works in DBs with transaction support)
BEGIN TRANSACTION;
DELETE FROM employees WHERE id = 2;
ROLLBACK;

-- SELECT for inserted data
SELECT * FROM employees;
