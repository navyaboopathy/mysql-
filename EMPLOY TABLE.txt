-- Step 1: Create Table
CREATE TABLE Employees (
    id INTEGER PRIMARY KEY,
    name TEXT NOT NULL,
    department TEXT,
    salary INTEGER,
    joining_date DATE
);

-- Step 2: Insert Sample Data
INSERT INTO Employees (id, name, department, salary, joining_date)
VALUES
(1, 'Alice', 'HR', 40000, '2022-01-10'),
(2, 'Bob', 'IT', 50000, '2021-03-15'),
(3, 'Charlie', 'Finance', 45000, '2020-07-22'),
(4, 'David', 'IT', 60000, '2023-02-01'),
(5, 'Eva', 'HR', 42000, '2022-11-12');

-- Step 3: SELECT * (Retrieve all data)
SELECT * FROM Employees;

-- Step 4: SELECT Specific Columns
SELECT name, salary FROM Employees;

-- Step 5: WHERE Clause
SELECT * FROM Employees WHERE department = 'IT';

-- Step 6: WHERE with AND
SELECT * FROM Employees WHERE department = 'IT' AND salary > 50000;

-- Step 7: WHERE with OR
SELECT * FROM Employees WHERE department = 'HR' OR salary < 45000;

-- Step 8: LIKE operator
SELECT * FROM Employees WHERE name LIKE 'D%';

-- Step 9: BETWEEN operator
SELECT * FROM Employees WHERE salary BETWEEN 40000 AND 50000;

-- Step 10: ORDER BY (Ascending)
SELECT * FROM Employees ORDER BY salary;

-- Step 11: ORDER BY (Descending)
SELECT * FROM Employees ORDER BY salary DESC;

-- Step 12: LIMIT
SELECT * FROM Employees ORDER BY salary DESC LIMIT 3;

-- Step 13: Alias
SELECT name AS Employee_Name, salary AS Monthly_Salary FROM Employees;

-- Step 14: DISTINCT
SELECT DISTINCT department FROM Employees;

-- Step 15: = vs IN
-- = Example:
SELECT * FROM Employees WHERE department = 'HR';

-- IN Example:
SELECT * FROM Employees WHERE department IN ('HR', 'Finance');
